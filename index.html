"Problem:
You are given a string (STR) of length N, consisting of only the lower case English alphabet.
Your task is to remove all the duplicate occurrences of characters in the string.

Input:
abcadeecfb

Output:
abcdef" 

      function removeDuplicates(str) {
        var uniqueChars = [];
        var result = "";
      
        for (var i = 0; i < str.length; i++) {
          var currentChar = str[i];

          if (!uniqueChars.includes(currentChar)) {
            uniqueChars.push(currentChar);
            result += currentChar;
          }
        }
      
        return result;
      }

      var input = "abcadeecfb";
      var output = removeDuplicates(input);
      console.log(output);

      function countAlphabets(str) {
        var charCountMap = new Map();

        for (var i = 0; i < str.length; i++) {
          var currentChar = str[i];

          if (charCountMap.has(currentChar)) {
            var count = charCountMap.get(currentChar);
            charCountMap.set(currentChar, count + 1);
          } else {
            charCountMap.set(currentChar, 1);
          }
        }

        charCountMap.forEach(function(value, key) {
          console.log(key + "=" + value);
        });
      }

      var input = "abcadeecfb";
      countAlphabets(input);
